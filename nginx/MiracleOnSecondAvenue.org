  server_name 
    www.miracleonsecondavenue.org
    miracleonsecondavenue.org
  ;

  root /var/www/MiracleOnSecondAvenue.org;

  include conf.d/common.conf;

  listen 80; # managed by Certbot
  listen 443 ssl; # managed by Certbot

  # Redirect non-https traffic to https
  if ($scheme != "https") {
    return 301 https://$host$request_uri;
  } # managed by Certbot

  ssl_certificate /etc/letsencrypt/live/devaprastha.info-0001/fullchain.pem; # managed by Certbot
  ssl_certificate_key /etc/letsencrypt/live/devaprastha.info-0001/privkey.pem; # managed by Certbot
  include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
  ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot

}
